47 void Gnomonic::Reverse(real lat0, real lon0, real x, real y,
48 real& lat, real& lon, real& azi, real& rk)
49 const {

51 azi0 = Math::atan2d(x, y),
52 rho = Math::hypot(x, y),
53 s = _a * atan(rho/_a);
54 bool little = rho <= _a;
55 if (!little)
56 rho = 1/rho;
57 GeodesicLine line(_earth.Line(lat0, lon0, azi0,
58 Geodesic::LATITUDE | Geodesic::LONGITUDE |
59 Geodesic::AZIMUTH | Geodesic::DISTANCE_IN |
60 Geodesic::REDUCEDLENGTH |
61 Geodesic::GEODESICSCALE));
62 int count = numit_, trip = 0;
63 real lat1, lon1, azi1, M;
64 while (count--) {
65 real m, t;
66 line.Position(s, lat1, lon1, azi1, m, M, t);
67 if (trip)
68 break;
69 // If little, solve rho(s) = rho with drho(s)/ds = 1/M^2
70 // else solve 1/rho(s) = 1/rho with d(1/rho(s))/ds = -1/m^2
71 real ds = little ? (m/M - rho) * M * M : (rho - M/m) * m * m;
72 s -= ds;
73 // Reversed test to allow escape with NaNs
74 if (!(abs(ds) >= eps_ * _a))
75 ++trip;
76 }
77 if (trip) {
78 lat = lat1; lon = lon1; azi = azi1; rk = M;
79 } else
80 lat = lon = azi = rk = Math::NaN();

82 }
